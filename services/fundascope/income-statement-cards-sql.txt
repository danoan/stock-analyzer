  1. Revenue Growth

  CAGR of Total Revenue from oldest to newest year.

  -- revenue_growth_grade.sql
  WITH bounds AS (
      SELECT
          MIN(year)          AS oldest_year,
          MAX(year)          AS newest_year,
          COUNT(DISTINCT year) AS n
      FROM income_stmt
      WHERE ticker = :ticker
        AND total_revenue > 0
  ),
  cagr AS (
      SELECT
          POWER(
              (SELECT total_revenue FROM income_stmt WHERE ticker = :ticker AND year = b.newest_year) /
              (SELECT total_revenue FROM income_stmt WHERE ticker = :ticker AND year = b.oldest_year),
              1.0 / (b.n - 1)
          ) - 1 AS value
      FROM bounds b
      WHERE b.n >= 2
  )
  SELECT
      ROUND(value * 100, 1)           AS revenue_cagr_pct,
      CASE
          WHEN value >  0.10 THEN 'A'
          WHEN value >  0.05 THEN 'B'
          WHEN value >  0.00 THEN 'C'
          WHEN value > -0.05 THEN 'D'
          ELSE 'F'
      END                              AS grade
  FROM cagr;

  Note: POWER() requires SQLite to be compiled with the math extension, or registered as a custom function via Python's sqlite3.create_function.

  2. Profit Trend

  Fraction of year-over-year periods where Net Income increased.

  -- profit_trend_grade.sql
  WITH series AS (
      SELECT
          year,
          net_income,
          LAG(net_income) OVER (ORDER BY year ASC) AS prev_ni
      FROM income_stmt
      WHERE ticker = :ticker
        AND net_income IS NOT NULL
  ),
  trend AS (
      SELECT
          AVG(CASE WHEN net_income > prev_ni THEN 1.0 ELSE 0.0 END) AS increase_ratio
      FROM series
      WHERE prev_ni IS NOT NULL     -- excludes first row (no previous)
  )
  SELECT
      ROUND(increase_ratio * 100, 1)  AS pct_years_increasing,
      CASE
          WHEN increase_ratio >= 0.66 THEN 'Improving'
          WHEN increase_ratio >= 0.33 THEN 'Mixed'
          ELSE                             'Declining'
      END                              AS direction,
      CASE
          WHEN increase_ratio >= 0.66 THEN 'A'
          WHEN increase_ratio >= 0.33 THEN 'C'
          ELSE                             'F'
      END                              AS grade
  FROM trend;

  3. Margin Health

  Average Operating Margin (Operating Income / Total Revenue) across all years.

  -- margin_health_grade.sql
  WITH margins AS (
      SELECT
          AVG(operating_income * 1.0 / NULLIF(total_revenue, 0)) AS avg_margin
      FROM income_stmt
      WHERE ticker = :ticker
        AND total_revenue IS NOT NULL
        AND total_revenue != 0
        AND operating_income IS NOT NULL
  )
  SELECT
      ROUND(avg_margin * 100, 1)   AS avg_operating_margin_pct,
      CASE
          WHEN avg_margin >  0.15 THEN 'Strong'
          WHEN avg_margin >  0.08 THEN 'Moderate'
          WHEN avg_margin >  0.00 THEN 'Weak'
          ELSE                         'Negative'
      END                           AS label,
      CASE
          WHEN avg_margin >  0.15 THEN 'A'
          WHEN avg_margin >  0.08 THEN 'B'
          WHEN avg_margin >  0.00 THEN 'D'
          ELSE                         'F'
      END                           AS grade
  FROM margins;

  4. Earnings Quality

  All net incomes positive, plus stability measured by coefficient of variation (population std dev / |mean|). High = all positive AND CV < 0.4.

  -- earnings_quality_grade.sql
  WITH stats AS (
      SELECT
          MIN(net_income)                                                    AS min_ni,
          COUNT(*)                                                           AS n,
          AVG(net_income)                                                    AS mean_ni,
          -- population std dev: sqrt(E[x²] - E[x]²)
          SQRT(AVG(net_income * net_income) - AVG(net_income) * AVG(net_income)) AS stddev_ni
      FROM income_stmt
      WHERE ticker = :ticker
        AND net_income IS NOT NULL
  ),
  cv AS (
      SELECT
          min_ni,
          n,
          CASE WHEN mean_ni = 0 THEN NULL
               ELSE stddev_ni / ABS(mean_ni)
          END AS coeff_variation
      FROM stats
  )
  SELECT
      ROUND(coeff_variation, 3)    AS cv,
      CASE
          WHEN min_ni > 0 AND n >= 2 AND coeff_variation < 0.4 THEN 'High'
          WHEN min_ni > 0                                        THEN 'Moderate'
          ELSE                                                        'Low'
      END                           AS quality,
      CASE
          WHEN min_ni > 0 AND n >= 2 AND coeff_variation < 0.4 THEN 'A'
          WHEN min_ni > 0                                        THEN 'B'
          ELSE                                                        'D'
      END                           AS grade
  FROM cv;

  5. Overall Grade (combined)

  Numeric average of the four grades above (A=5 B=4 C=3 D=2 F=1), converted back to a letter.

  -- overall_grade.sql
  WITH
  -- Revenue Growth
  bounds AS (
      SELECT MIN(year) AS oy, MAX(year) AS ny, COUNT(DISTINCT year) AS n
      FROM income_stmt WHERE ticker = :ticker AND total_revenue > 0
  ),
  revenue_grade AS (
      SELECT CASE
          WHEN POWER((SELECT total_revenue FROM income_stmt WHERE ticker=:ticker AND year=b.ny) /
                     (SELECT total_revenue FROM income_stmt WHERE ticker=:ticker AND year=b.oy),
                     1.0/(b.n-1)) - 1 >  0.10 THEN 5
          WHEN POWER((SELECT total_revenue FROM income_stmt WHERE ticker=:ticker AND year=b.ny) /
                     (SELECT total_revenue FROM income_stmt WHERE ticker=:ticker AND year=b.oy),
                     1.0/(b.n-1)) - 1 >  0.05 THEN 4
          WHEN POWER((SELECT total_revenue FROM income_stmt WHERE ticker=:ticker AND year=b.ny) /
                     (SELECT total_revenue FROM income_stmt WHERE ticker=:ticker AND year=b.oy),
                     1.0/(b.n-1)) - 1 >  0.00 THEN 3
          WHEN POWER((SELECT total_revenue FROM income_stmt WHERE ticker=:ticker AND year=b.ny) /
                     (SELECT total_revenue FROM income_stmt WHERE ticker=:ticker AND year=b.oy),
                     1.0/(b.n-1)) - 1 > -0.05 THEN 2
          ELSE 1
      END AS score
      FROM bounds b WHERE b.n >= 2
  ),
  -- Profit Trend
  ni_series AS (
      SELECT net_income, LAG(net_income) OVER (ORDER BY year) AS prev_ni
      FROM income_stmt WHERE ticker = :ticker AND net_income IS NOT NULL
  ),
  profit_grade AS (
      SELECT CASE
          WHEN AVG(CASE WHEN net_income > prev_ni THEN 1.0 ELSE 0.0 END) >= 0.66 THEN 5
          WHEN AVG(CASE WHEN net_income > prev_ni THEN 1.0 ELSE 0.0 END) >= 0.33 THEN 3
          ELSE 1
      END AS score
      FROM ni_series WHERE prev_ni IS NOT NULL
  ),
  -- Margin Health
  margin_grade AS (
      SELECT CASE
          WHEN AVG(operating_income / NULLIF(total_revenue, 0)) >  0.15 THEN 5
          WHEN AVG(operating_income / NULLIF(total_revenue, 0)) >  0.08 THEN 4
          WHEN AVG(operating_income / NULLIF(total_revenue, 0)) >  0.00 THEN 2
          ELSE 1
      END AS score
      FROM income_stmt
      WHERE ticker = :ticker AND total_revenue IS NOT NULL AND total_revenue != 0
  ),
  -- Earnings Quality
  eq_stats AS (
      SELECT MIN(net_income) AS min_ni, COUNT(*) AS n,
             SQRT(AVG(net_income*net_income) - AVG(net_income)*AVG(net_income))
               / ABS(NULLIF(AVG(net_income), 0)) AS cv
      FROM income_stmt WHERE ticker = :ticker AND net_income IS NOT NULL
  ),
  earnings_grade AS (
      SELECT CASE
          WHEN min_ni > 0 AND n >= 2 AND cv < 0.4 THEN 5
          WHEN min_ni > 0                           THEN 4
          ELSE 2
      END AS score
      FROM eq_stats
  )
  SELECT
      CASE
          WHEN (rg.score + pg.score + mg.score + eg.score) / 4.0 >= 4.5 THEN 'A'
          WHEN (rg.score + pg.score + mg.score + eg.score) / 4.0 >= 3.5 THEN 'B'
          WHEN (rg.score + pg.score + mg.score + eg.score) / 4.0 >= 2.5 THEN 'C'
          WHEN (rg.score + pg.score + mg.score + eg.score) / 4.0 >= 1.5 THEN 'D'
          ELSE 'F'
      END AS overall_grade
  FROM revenue_grade rg, profit_grade pg, margin_grade mg, earnings_grade eg;
